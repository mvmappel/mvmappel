namespace NinjaTrader.NinjaScript.Indicators
{
	public class MAs3EMA : Indicator
	{
		private const int totalObjs = 8;
		private int upCnt = 0;
		private int downCnt = 0;
		private int crossFast = 0;
		private int crossSlow = 0;
		private int slope_period = 14;
		private int std_period = 40;
		private double std_mult = 1.0;
		
		protected override void OnStateChange()
		{
			if (State == State.SetDefaults)
			{
				Description									= @"Enter the description for your new custom Indicator here.";
				Name										= "MAs3EMA";
				Calculate									= Calculate.OnBarClose;
				IsOverlay									= true;
				DisplayInDataBox							= true;
				DrawOnPricePanel							= true;
				DrawHorizontalGridLines						= true;
				DrawVerticalGridLines						= true;
				PaintPriceMarkers							= true;
				ScaleJustification							= NinjaTrader.Gui.Chart.ScaleJustification.Right;
				//Disable this property if your indicator requires custom values that cumulate with each new market data event. 
				//See Help Guide for additional information.
				IsSuspendedWhileInactive					= true;
				Cross3EMAs 									= true;
				EmaFast										= 10;
				EmaMiddle									= 21;
				EmaSlow										= 55;
				EmaFast2									=  8;
				EmaSlow2									= 21;
				AddPlot(Brushes.Green, "EmaFastPlot");
				AddPlot(Brushes.Orange, "EmaMiddlePlot");
				AddPlot(Brushes.Red, "EmaSlowPlot");
				AddPlot(Brushes.Green, "EmaFastPlot2");
				AddPlot(Brushes.Red, "EmaSlowPlot2");
			}
			else if (State == State.Configure)
			{
				//AddChartIndicator(LinRegSlopeBounds(slope_period,std_period,std_mult));
			}
		}

		protected override void OnBarUpdate()
		{
			ShowADXSlope();
			
			if (Cross3EMAs) {
				double emaFast = EMA(Close,EmaFast)[0];
				double emaMiddle = EMA(Close,EmaMiddle)[0];
				double emaSlow = EMA(Close,EmaSlow)[0];
				crossFast = EmaMiddle;
				crossSlow = EmaSlow;
				
				EmaFastPlot[0] = emaFast;
				EmaMiddlePlot[0] = emaMiddle;
				EmaSlowPlot[0] = emaSlow;
			} else {
				double emaFast2 = EMA(Close,EmaFast2)[0];
				double emaSlow2 = EMA(Close,EmaSlow2)[0];
				crossFast = EmaFast2;
				crossSlow = EmaSlow2;
				
				EmaFastPlot2[0] = emaFast2;
				EmaSlowPlot2[0] = emaSlow2;				
			}
			
			if (CrossAbove(EMA(Close,crossFast), EMA(Close,crossSlow), 1)) {
				string arrowUpName = CreateObjName("MAsUpArrow", upCnt);
				if (upCnt < totalObjs) {
					upCnt++;
				} else {
					upCnt = 0;
				}
				
				Draw.ArrowUp(this, arrowUpName, true, 0, EMA(Close,EmaSlow)[0] - 22 * TickSize, Brushes.Green);
				//Draw.Region(this, "MAsUpCloud", CurrentBar, 0, EMA(Close,EmaSlow), EMA(Close,EmaFast), null, Brushes.Blue, 1);
				PlaySound(NinjaTrader.Core.Globals.InstallDir + @"\sounds\Alert1.wav");
			}
			
			if (CrossBelow(EMA(Close,crossFast), EMA(Close,crossSlow), 1)) {			
				string arrowDownName = CreateObjName("MAsDownArrow", downCnt);				
				if (downCnt < totalObjs) {
					downCnt++;
				} else {
					downCnt = 0;
				}
				
				Draw.ArrowDown(this, arrowDownName, true, 0, EMA(Close,EmaSlow)[0] + 22 * TickSize, Brushes.Red);
				PlaySound(NinjaTrader.Core.Globals.InstallDir + @"\sounds\Alert1.wav");
			}
		}

		private void ShowADXSlope()
		{		
		    // Ensure we have enough data
		    if (CurrentBar >= 14) {
			
			    // Get current ADX value
			    double adxValue = ADX(14)[0];
				
				double slope = LinRegSlopeBounds(slope_period,std_period,std_mult).Slope[0];
			
			    // Draw the value on the screen
			    TextFixed LblTxt = Draw.TextFixed(this, "ADXDisplay", "ADX: " + adxValue.ToString("0.00") + "  Slope: " + slope.ToString("0.00"), TextPosition.TopLeft);
				
				if(adxValue >= 25.00 && slope > 0) {
					LblTxt.TextBrush = Brushes.Green;
				} else {
					LblTxt.TextBrush = Brushes.Red;					
				}
				
				SimpleFont LblFont = new NinjaTrader.Gui.Tools.SimpleFont("Courier New", 12) { Size = 18, Bold = true };				
				LblTxt.Font = LblFont;
			}
		}			
		
		private string CreateObjName(string name, int iCnt)
		{		
			return name + "_" + iCnt;
		}		

		#region Properties

		{ get; set; }		
		
		[NinjaScriptProperty]
		[Range(1, int.MaxValue)]
		[Display(Name="EmaFast2", Order=5, GroupName="Parameters")]
		public int EmaFast2
		{ get; set; }

		[NinjaScriptProperty]
		[Range(1, int.MaxValue)]
		[Display(Name="EmaSlow2", Order=6, GroupName="Parameters")]
		public int EmaSlow2
		{ get; set; }		

		[Browsable(false)]
		[XmlIgnore]
		public Series<double> EmaFastPlot
		{
			get { return Values[0]; }
		}

		[Browsable(false)]
		[XmlIgnore]

		{
			get { return Values[1]; }
		}

		[Browsable(false)]

		}

		#endregion
	}
